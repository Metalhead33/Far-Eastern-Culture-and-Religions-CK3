create_male_slave = {
	if = {
		limit = {
			AND = {
				NOT = { $FAITH$ = { has_doctrine = tenet_unreformed_syncretism } }
				OR = {
					$OWNER$.culture = { has_cultural_parameter = can_castrate_prisoners }
					$OWNER$.faith = { religion_tag = islam_religion }
				}
			}
		}
		random_list = {
			20 = {
				create_character = {
					location = ROOT.capital_province
					faith = scope:slave_faith
					culture = scope:slave_culture
					template = eunuch_slave_character
					save_scope_as = new_courtier
				}
				scope:new_courtier = {
					random_list = {
						33 = { add_trait = intellect_good_1 }
						33 = { add_trait = intellect_good_2 }
						33 = { add_trait = intellect_good_3 }
					}
					random_list = {
						16 = {
							add_trait = administrator
						}
						16 = {
							add_trait = architect
						}
						16 = {
							add_trait = lifestyle_gardener
						}
						16 = {
							add_trait = reveler_1
						}
						16 = {
							add_trait = physician_1
						}
						8 = {
							add_trait = physician_2
						}
						4 = {
							add_trait = physician_3
						}
						16 = {
							add_trait = poet
						}
					}
					set_to_lowborn = yes
					# learn_language_of_culture = ROOT.culture
					trigger_event = { on_action = carn_character_initialization_pulse }
				}
			}
			80 = {
				create_character = {
					location = $OWNER$.capital_province
					template = male_slave_character
					culture = $CULTURE$
					faith = $FAITH$
					save_scope_as = new_courtier
				}
				scope:new_courtier = {
					random_list = {
						33 = { add_trait = physique_good_1 }
						33 = { add_trait = physique_good_2 }
						33 = { add_trait = physique_good_3 }
					}
				}
			}
		}
	}
	else = {
		create_character = {
			location = $OWNER$.capital_province
			template = male_slave_character
			culture = $CULTURE$
			faith = $FAITH$
			save_scope_as = new_courtier
		}
		scope:new_courtier = {
			random_list = {
				33 = { add_trait = physique_good_1 }
				33 = { add_trait = physique_good_2 }
				33 = { add_trait = physique_good_3 }
			}
		}
	}
	scope:new_courtier = {
		set_to_lowborn = yes
		learn_language_of_culture = $OWNER$.culture
		trigger_event = { on_action = carn_character_initialization_pulse }
	}
	if = {
		limit = {
			$OWNER$ = {
				OR = {
					has_trait = compassionate
					has_trait = just
				}
			}
		}
		carn_enslave_effect = {
			SLAVE = scope:new_courtier
			OWNER = $OWNER$
			DRAMA = no
		}
	}
	else = {
		carn_enslave_effect = {
			SLAVE = scope:new_courtier
			OWNER = $OWNER$
			DRAMA = yes
		}
	}
}

		random_list = {
			33 = { add_trait = beauty_good_1 }

		limit = {

		carn_enslave_effect = {
			SLAVE = scope:new_courtier
			OWNER = $OWNER$
			DRAMA = no
		}
	}
	else = {
		carn_enslave_effect = {
			SLAVE = scope:new_courtier
			OWNER = $OWNER$
			DRAMA = yes
		}
	}
}
