variable_add_effect = {
	if = {
		limit = {
			NOT = { has_variable = $VARIABLE$ }
		}
		set_variable = {
			name = $VARIABLE$
			value = 0
		}
	}
	change_variable = {
		name = $VARIABLE$
		add = $VALUE$
	}
}
